	public static String checkCondition(String newValue,String oldValue) {
		double newvalue = Double.parseDouble(newValue);
		double oldvalue = Double.parseDouble(oldValue.replace("not ok", "").replace("ok", ""));
		if(oldvalue>200.0) {
		double result=0.0;
		if(newvalue>oldvalue) {
		double diff=newvalue-oldvalue;
			result = diff/newvalue;
			result = result * 100;
		}
		if(result>30.0) {
			return Math.round(newvalue)+"not ok";
		}
		else {
				return Math.round(newvalue)+"ok";
		}	
		}
		else {
			double result=0.0;
			if(newvalue>oldvalue) {
			result=newvalue-oldvalue;
			}
			if(result>30.0) {
				return Math.round(newvalue)+"not ok";
			}
			else {
					return Math.round(newvalue)+"ok";
			}	
		}

	public com.jcraft.jsch.Session getConection(String host,String user,String password)throws Exception {
		    Properties config = new Properties();
			config.put("StrictHostKeyChecking", "no");
			JSch jsch = new JSch();
			// Create a JSch session to connect to the server
			com.jcraft.jsch.Session session = jsch.getSession(user, host, 22);
			session.setPassword(password);
			session.setConfig(config);
			// Establish the connection
			session.connect();
			System.out.println("Connected...");
			return session;
	}
	
	public String executeCommand(String command, com.jcraft.jsch.Session session) {
		try {
		ChannelExec channel = (ChannelExec) session.openChannel("exec");
		channel.setCommand(command);
		channel.setErrStream(System.err);
        String response=null;
		InputStream in = channel.getInputStream();
		channel.connect();
		byte[] tmp = new byte[1024];
		while (true) {
			while (in.available() > 0) { 
				int i = in.read(tmp, 0, 1024);
				if (i < 0) {
					break;
				}
				response=response+new String(tmp, 0, i);
				System.out.print(new String(tmp, 0, i));
			}
			if (channel.isClosed()) {
				System.out.println("Exit Status: "
						+ channel.getExitStatus());
				break;
			}
			Thread.sleep(1000);
			System.out.println("Keshav");
		}
		String res=StringUtils.substringBetween(response, "min/avg/max/stddev = "," ms");
		String[] words=res.split("/");
		System.out.println(words[1]);
		channel.disconnect();
	    return words[1];
		}catch(Exception e)
		{
			return "0.0";
		}
	}
	
	public void disconect(com.jcraft.jsch.Session session)
	{
		session.disconnect();
		System.out.println("disconnected..");
	}
